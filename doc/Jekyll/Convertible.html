def post_render
  if self.site.post_filters and is_filterable?
    self.site.post_filters.each do |filter|
      filter.post_render(self)
    end
  end
end

def pre_render
  self.site.load_post_filters unless self.site.post_filters

  if self.site.post_filters and is_filterable?
    self.site.post_filters.each do |filter|
      filter.pre_render(self)
    end
  end
end
