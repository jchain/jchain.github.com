def render(context)
  output = super
  if output =~ %r\{"\s*(.+?)\s*"\}/
    @quote = RubyPants.new($1).to_html
    "<span class='pullquote-#{@align}' data-pullquote='#{@quote}'>#{output.gsub(/\{"\s*|\s*"\}/, '')}</span>"
  else
    return "Surround your pullquote like this {\" text to be quoted \"}"
  end
end
