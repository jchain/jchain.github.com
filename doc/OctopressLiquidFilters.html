    # File plugins/octopress_filters.rb, line 79
def expand_urls(input, url='')
  url ||= '/'
  input.gsub %r(\s+(href|src)\s*=\s*["|']{1})(\/[^\"'>]*)/ do
    $1+url+$3
  end
end

    raw_content(input) click to toggle source

    Extracts raw content DIV from template, used for page description as {{ content }} contains complete sub-template code on main page level

    # File plugins/octopress_filters.rb, line 68
def raw_content(input)
  %r<div class="entry-content">(?<content>[\s\S]*?)<\/div>\s*<(footer|\/article)>/ =~ input
  return (content.nil?) ? input : content
end

    truncate(input, length) click to toggle source

    Improved version of Liquid’s truncate:

    • Doesn’t cut in the middle of a word.

    • Uses typographically correct ellipsis (…) insted of ‘…’

    # File plugins/octopress_filters.rb, line 89
def truncate(input, length)
  if input.length > length && input[0..(length-1)] =~ %r(.+)\b.+$/m
    $1.strip + ' &hellip;'
  else
    input
  end
end

    Improved version of Liquid’s truncatewords:

    • Uses typographically correct ellipsis (…) insted of ‘…’

    # File plugins/octopress_filters.rb, line 99
def truncatewords(input, length)
  truncate = input.split(' ')
  if truncate.length > length
    truncate[0..length-1].join(' ').strip + ' &hellip;'
  else
    input
  end
end
